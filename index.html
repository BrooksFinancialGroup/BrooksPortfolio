# Import necessary libraries import matplotlib.pyplot as plt # Function to plot Freedom Strategies def plot_freedom_strategies(): # Strategies and their associated 5-Year Annualized Returns (Example Data) strategies = ['Aggressive', 'Growth', 'Moderate Growth', 'Balanced', 'Conservative', 'Defensive'] returns = [5.7, 6.4, 6.0, 6.1, 6.3, 6.7] # Use the accurate data as necessary # Create a bar chart plt.figure(figsize=(10, 5)) plt.bar(strategies, returns, color='lightblue') plt.title('Freedom Foundation 5-Year Annualized Returns') plt.xlabel('Strategies') plt.ylabel('Returns (%)') plt.xticks(rotation=45) plt.tight_layout() # Save the figure plt.savefig('freedom_strategies.png') plt.show() # Execute the plotting function if __name__ == "__main__": plot_freedom_strategies()
